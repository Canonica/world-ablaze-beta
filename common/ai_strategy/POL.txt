british_help_against_germany = {
	enable = {
		tag = POL
		OR = {
			GER = { has_completed_focus = GER_second_vienna_award }
			GER = { has_war_with = POL }
		}
		country_exists = ENG
	}
	abort = {
		NOT = { country_exists = ENG }
	}
	
	ai_strategy = {
		type = befriend
		id = "ENG"			
		value = 200
	}
	ai_strategy = {
		type = alliance
		id = "ENG"			
		value = 200
	}	
}
