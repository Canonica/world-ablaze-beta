EAI_NAV_aircraft_production_strategy_2 = {
	enable = {
		WA_has_nav_tech = yes
		
		num_of_military_factories > 40

		has_deployed_air_force_size = { size < 2000 type = naval_bomber }
	}
	abort = {
		NOT = {
			AND = {
				WA_has_nav_tech = yes
		
				num_of_military_factories > 40

				has_deployed_air_force_size = { size < 2500 type = naval_bomber }
			}
		}
	}

	ai_strategy = { type = equipment_variant_production_factor id = cv_nav_bomber_equipment  value = 2 }
	ai_strategy = { type = equipment_production_min_factories id = naval_bomber         value = 3 }
}

EAI_NAV_aircraft_production_strategy_3 = {
	enable = {
		WA_has_nav_tech = yes

		has_deployed_air_force_size = { size < 2000 type = naval_bomber }
		
		num_of_military_factories > 50

		alliance_naval_strength_ratio < 1
	}
	abort = {
		NOT = {
			AND = {
				WA_has_nav_tech = yes

				has_deployed_air_force_size = { size < 2500 type = naval_bomber }
		
				num_of_military_factories > 50
			}
		}
	}

	ai_strategy = { type = equipment_production_min_factories id = naval_bomber         value = 2 }
}
