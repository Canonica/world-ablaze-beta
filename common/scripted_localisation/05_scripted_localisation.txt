# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = { # Country
	name = GetSOVinfiltrationlevel
	text = {
		trigger = {
			check_variable = { SOV_infiltration_level > 99 }
		}
		localization_key = SOV_infiltration_level_Overpowering
	}
	text = {
		trigger = {
			check_variable = { SOV_infiltration_level > 74 }
		}
		localization_key = SOV_infiltration_level_Large
	}
	text = {
		trigger = {
			check_variable = { SOV_infiltration_level > 49 }
		}
		localization_key = SOV_infiltration_level_Moderate
	}
	text = {
		trigger = {
			check_variable = { SOV_infiltration_level > 24 }
		}
		localization_key = SOV_infiltration_level_Small
	}
	text = {
		trigger = {
			check_variable = { SOV_infiltration_level > 0 }
		}
		localization_key = SOV_infiltration_level_miniscule
	}
	text = {
		localization_key = SOV_infiltration_level_Zero
	}
}
