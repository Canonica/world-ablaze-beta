		tag = PRC
		is_ai = yes
		has_war_With = JAP
		621 = { is_controlled_by = PRC }

		OR = {
			NOT = { has_full_control_of_state = 621 }

		add_equipment_to_stockpile = { type = artillery_equipment_1 amount = 1000 producer = PRC }
		add_political_power = 1000
	}

	option = {}
}

		tag = PRC

		NOT = { 609 = { is_controlled_by = JAP } }
